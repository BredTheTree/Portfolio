Arduino DC Motor Controller

IR Remote Controlled DC Motor with Arduino

C++ Arduino Electronics IR Remote

📋 Project Overview

Summary

This Arduino project is a DC motor wired to an IR sensor, allowing its speed and direction to be controlled using a remote control

Key Features:

  • A DC motor capable of switching speed and direction at the press of a button
  • An IR sensor to pickup inputs from the remote
  • An LCD screen that displays the current speed and direction of the motor
  • A 9V battery to provide enough power to the circuit

Technical Specifications:

Microcontroller:
Arduino Uno R3
Motor Driver:
L293D
Motor Type:
9v DC motor with fan attatchment
Power Supply:
9v battery w/ power supply module + laptop
IR Receiver:
VS1838

💻 Source Code

arduino_motor_controller.ino
// Force IRremote to use Timer1 (avoids LCD timing conflict) #define IR_USE_TIMER1 #include <IRremote.hpp> #include <LiquidCrystal.h> // LCD pin setup LiquidCrystal lcd(12, 13, 7, 8, 9, 10); // Motor pins #define RECV_PIN 11 #define SPEED_PIN 5 #define DIR1 4 #define DIR2 3 // Motor state int mSpeed = 0; // Current motor speed (0–255) bool forward = true; // Direction: true = forward int lastSpeed = -1; bool lastDirection = false; void setup() { // Initialize LCD lcd.begin(16, 2); lcd.print("IR Motor Ready"); // Serial for debug Serial.begin(9600); Serial.println("IR Motor Control Active"); // IR setup IrReceiver.begin(RECV_PIN, ENABLE_LED_FEEDBACK); // Motor pins pinMode(SPEED_PIN, OUTPUT); pinMode(DIR1, OUTPUT); pinMode(DIR2, OUTPUT); } void loop() { // Check IR input if (IrReceiver.decode()) { unsigned long code = IrReceiver.decodedIRData.decodedRawData; if (code != 0) { // Ignore spurious zeros Serial.print("Code received: "); Serial.println(code, HEX); switch (code) { case 0xBA45FF00: // Power - stop motor mSpeed = 0; Serial.println("Motor stopped"); break; case 0xB946FF00: // Vol+ - increase speed mSpeed += 5; if (mSpeed > 255) mSpeed = 255; if (mSpeed < 155 && mSpeed != 0) mSpeed = 155; // prevent stall Serial.print("Speed increased to "); Serial.println(mSpeed); break; case 0xEA15FF00: // Vol- - decrease speed mSpeed -= 5; if (mSpeed < 155 && mSpeed != 0) mSpeed = 155; // prevent stall Serial.print("Speed decreased to "); Serial.println(mSpeed); break; case 0xBF40FF00: // Direction toggle forward = !forward; Serial.print("Direction: "); Serial.println(forward ? "Forward" : "Reverse"); break; default: Serial.println("Unknown button"); break; } } IrReceiver.resume(); // Prepare for next code } // Apply motor direction digitalWrite(DIR1, forward ? HIGH : LOW); digitalWrite(DIR2, forward ? LOW : HIGH); analogWrite(SPEED_PIN, mSpeed); // Update LCD only when speed/direction changes if (mSpeed != lastSpeed || forward != lastDirection) { noInterrupts(); // prevent LCD corruption lcd.setCursor(0, 0); lcd.print("Speed: "); // pad with spaces lcd.setCursor(7, 0); lcd.print(mSpeed); lcd.setCursor(0, 1); lcd.print("Dir: "); // pad with spaces lcd.setCursor(5, 1); lcd.print(forward ? "Forward" : "Reverse"); interrupts(); lastSpeed = mSpeed; lastDirection = forward; } }
